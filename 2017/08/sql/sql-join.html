SQLдёӯinner joinгҖҒouter joinе’Ңcross joinзҡ„еҢәеҲ«

INNER JOIN дә§з”ҹзҡ„з»“жһңжҳҜABзҡ„дәӨйӣҶ

SELECT * FROM TableA INNER JOIN TableB ON TableA.name = TableB.name

LEFT [OUTER] JOIN дә§з”ҹиЎЁAзҡ„е®Ңе…ЁйӣҶпјҢиҖҢBиЎЁдёӯеҢ№й…Қзҡ„еҲҷжңүеҖјпјҢжІЎжңүеҢ№й…Қзҡ„еҲҷд»ҘnullеҖјеҸ–д»Ј

SELECT * FROM TableA LEFT OUTER JOIN TableB ON TableA.name = TableB.name

RIGHT [OUTER] JOIN дә§з”ҹиЎЁBзҡ„е®Ңе…ЁйӣҶпјҢиҖҢAиЎЁдёӯеҢ№й…Қзҡ„еҲҷжңүеҖјпјҢжІЎжңүеҢ№й…Қзҡ„еҲҷд»ҘnullеҖјеҸ–д»Ј

SELECT * FROM TableA RIGHT OUTER JOIN TableB ON TableA.name = TableB.name

FULL [OUTER] JOIN дә§з”ҹAе’ҢBзҡ„е№¶йӣҶгҖӮеҜ№дәҺжІЎжңүеҢ№й…Қзҡ„и®°еҪ•пјҢеҲҷдјҡд»ҘnullеҒҡдёәеҖј

SELECT FROM TableA FULL OUTER JOIN TableB ON TableA.name = TableB.name
дҪ еҸҜд»ҘйҖҡиҝҮis NULLе°ҶжІЎжңүеҢ№й…Қзҡ„еҖјжүҫеҮәжқҘпјҡ
SELECT FROM TableA FULL OUTER JOIN TableB ON TableA.name = TableB.name
WHERE TableA.id IS null OR TableB.id IS null

CROSS JOIN жҠҠиЎЁAе’ҢиЎЁBзҡ„ж•°жҚ®иҝӣиЎҢдёҖдёӘNMзҡ„з»„еҗҲпјҢеҚіз¬ӣеҚЎе°”з§ҜгҖӮеҰӮжң¬дҫӢдјҡдә§з”ҹ44=16жқЎи®°еҪ•пјҢеңЁејҖеҸ‘иҝҮзЁӢдёӯжҲ‘д»¬иӮҜе®ҡжҳҜиҰҒиҝҮж»Өж•°жҚ®пјҢжүҖд»Ҙиҝҷз§ҚеҫҲе°‘з”Ё

SELECT * FROM TableA CROSS JOIN TableB
